294  PCLNormalPtr normalCluster = PCManager::estimateNormal( cluster);
295 
296  // initialize inlier count
297  int planeInl = 0, coneInl = 0, sphereInl = 0, cylinderInl = 0;
298  int clusterSize = cluster->size();
299 
300  // call ransac sphere service

302  if( callRansacSphereSegmentation( cluster, normalCluster, outSphere))
303  sphereInl = outSphere->response.inliers.size();
304 
305  // call ransac cylinder service
306  PrimitiveSegmentationPtr outCylinder ( new PrimitiveSegmentation);
307  if( callRansacCylinderSegmentation( cluster, normalCluster, outCylinder))
308  cylinderInl = outCylinder->response.inliers.size();
309 
310 
311  // call ransac cone service
312  PrimitiveSegmentationPtr outCone ( new PrimitiveSegmentation);
313  if( callRansacConeSegmentation( cluster, normalCluster, outCone))
314  coneInl = outCone->response.inliers.size();
315 
316  // call ransac plane service
317  PrimitiveSegmentationPtr outPlane ( new PrimitiveSegmentation);
318  if( callRansacPlaneSegmentation( cluster, normalCluster, outPlane))
319  planeInl = outPlane->response.inliers.size();
320 
321  // show cluster with color w.r.t. primitive
322  int R, G, B;

344  }else if( ( planeInl >= coneInl) && ( planeInl >= sphereInl) && ( planeInl >= cylinderInl)){
345  R = 255; G = 0; B = 0; // plane : RED
